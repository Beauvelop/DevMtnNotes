----------call(); method is actually pretty simple It tells the function what '.this' is!---------


Here we'll have call(); --with one arg
var breakfast = {
meal: "eggs & coffee",
time: "8:00am",
};
var lunch = {
meal: "a sandwich and a beer",
time: "noon",
};
var dinner = {
meal: "roast chicken with wine",
time: "9:00pm",
};
function mealPlan() {
document.write("At around " + this.time +
" we're probably going to have " + this.meal +".");
}
mealPlan.call(lunch);

// At around noon we're probably going to have a sandwich and a beer.

--------------------Here we'll have call(); --with two args

First, let's make mealPlan() take an argument...
var breakfast = {
meal: "eggs & coffee",
time: "8:00am",
};
var lunch = {
meal: "a sandwich and a beer",
time: "noon",
};
var dinner = {
meal: "roast chicken with wine",
time: "9:00pm",
};
function mealPlan(location) {
document.write("At around " + this.time +
" we're probably going to have " + this.meal +". Why don't we meet at " + location + "?");
}
mealPlan.call(dinner, "Denny's");

// At around 9:00pm we're probably going to have roast chicken with wine. Why don't we meet at Denny's?

  • So 1st arg of call()--determines what is '.this'.

  • 2nd arg is the parameter fed to the function referred to by that which the .call() is placed!


----------APPLY(); method can perform an action on multiple functions----------


var film = {
star: "Matt Damon",
title: "Bourne",
};

function review(starNumber) {
document.write("The film " + this.title + " stars " + this.star +
" and received "+ starNumber + " out of 5 stars.");
}
function suggest(activity, film, args){
activity.apply(args);
}
suggest(review, film, [2]);
The film Bourne stars Matt Damon and received 2 out of 5 stars.
    Args in the apply(); method:
  1. name of function to be invoked
  2. used as '.this'
  3. args as Array passed to the function
